00106                   _RAIter1 __end1,
00107                   _RAIter2 __begin2, _Pred __pred)
00108       {
00109         // Passed end iterator is one short.
00110         _RAIter1 __spot = adjacent_find(__begin1, __end1 + 1,
00111                     __pred, sequential_tag());
