00152 PB_DS_CLASS_T_DEC
00153 void
00154 PB_DS_CLASS_C_DEC::
00155 calc_max_num_coll()
00156 {
00157   // max_col <-- \sqrt{2 load \ln( 2 m \ln( m ) ) }
00158   const double ln_arg = 2 * m_size * std::log(double(m_size));
00159   m_max_col = size_type(std::ceil(std::sqrt(2 * m_load * std::log(ln_arg))));
00160 
00161 #ifdef PB_DS_HT_MAP_RESIZE_TRACE_
00162   std::cerr << "chmccrt::calc_max_num_coll " 
00163         << static_cast<unsigned long>(m_size) <<    "    " 
00164         << static_cast<unsigned long>(m_max_col) << std::endl;
00165 #endif 
00166 }
