00099   /**
00100    *  @brief A generalization of pointer arithmetic.
00101    *  @param  __first  An input iterator.
00102    *  @param  __last  An input iterator.
00103    *  @return  The distance between them.
00104    *
00105    *  Returns @c n such that __first + n == __last.  This requires
00106    *  that @p __last must be reachable from @p __first.  Note that @c
00107    *  n may be negative.
00108    *
00109    *  For random access iterators, this uses their @c + and @c - operations
00110    *  and are constant time.  For other %iterator classes they are linear time.

00113     inline typename iterator_traits<_InputIterator>::difference_type
00114     distance(_InputIterator __first, _InputIterator __last)
00115     {
00116       // concept requirements -- taken care of in __distance
